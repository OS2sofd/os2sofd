mysql -u kommunenavn -h aurora-cluster.cluster-cmff5r8k4jr5.eu-west-1.rds.amazonaws.com -p --database os2rollekatalog_kommunenavn -e "select ou.uuid as UUID, ifnull(roles.hasRoles,'Nej') as 'Roller tilknyttet', replace(replace(ou.name,'\t',''),';','') as Enhed, replace(replace(pou1.name,'\t',''),';','') as 'Overliggende enhed','' as 'Kaldenavn Kort i LOS' from ous ou left join ous pou1 ON pou1.uuid = ou.parent_uuid left join (select ou_uuid, 'Ja' as hasRoles FROM ou_roles group by ou_uuid) as roles ON roles.ou_uuid = ou.uuid where ou.active = 1" | sed 's/\t/;/g' > AD_LOS_Mapping.csv
